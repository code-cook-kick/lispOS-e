Contract Validity Checker Code:

function checkContractValidity(hasConsent, hasObject, hasCause) { // Legal basis: Article 1318, Civil Code of the Philippines // All three elements must be present for a valid contract const isValid = hasConsent && hasObject && hasCause; // Identify missing elements const missingElements = []; if (!hasConsent) missingElements.push('Consent'); if (!hasObject) missingElements.push('Object'); if (!hasCause) missingElements.push('Cause'); return { valid: isValid, status: isValid ? 'VALID CONTRACT' : 'INVALID CONTRACT', missingElements: missingElements, legalEffect: isValid ? 'Contract is enforceable' : 'No contract exists - remedy missing elements' }; }
